```
#include<Windows.h>
#include<WinDef.h>
#include<wchar.h>
void HandleSizeChange(HWND hWnd, UINT state, int width, int height);

LRESULT CALLBACK WindowProc(HWND hWnd, UINT uMsg, WPARAM wParam, LPARAM lParam)
{
	switch (uMsg)
	{
	case WM_SIZE:
	{
		UINT state = (UINT)wParam;
		int width = LOWORD(lParam);
		int height = HIWORD(lParam);
		HandleSizeChange(hWnd, state, width, height);
		return 0;
	}
	case WM_DESTROY:
		PostQuitMessage(0);
		return 0;

	default:
		return DefWindowProc(hWnd, uMsg, wParam, lParam);
	}
}
int WINAPI WinMain(HINSTANCE hInstance, HINSTANCE hPrevInstance, LPSTR lpCmdLine, int nShowCmd)
	{
		WNDCLASSEXA wndcls = { 0 };
		wndcls.cbSize = sizeof(WNDCLASSEXA);
		wndcls.style = CS_HREDRAW | CS_VREDRAW;
		wndcls.lpfnWndProc = WindowProc;
		wndcls.hInstance = hInstance;
		wndcls.hIcon = LoadIcon(NULL, IDI_APPLICATION);
		wndcls.hCursor = LoadCursor(NULL, IDC_ARROW);
		wndcls.hbrBackground = (HBRUSH)(COLOR_WINDOW + 1);
		wndcls.lpszClassName = "MyWindowClass";

		if (!RegisterClassExA(&wndcls))
		{
			MessageBoxA(NULL, "窗口类注册失败!", "错误", MB_ICONERROR);
			return 1;
		}

		HWND hwnd = CreateWindowExA(
			0,
			"MyWindowClass",
			"1",
			WS_OVERLAPPEDWINDOW,
			CW_USEDEFAULT, CW_USEDEFAULT,
			800, 600,
			NULL,
			NULL,
			hInstance,
			NULL
		);

		if (!hwnd)
		{
			MessageBoxA(NULL, "窗口创建失败!", "错误", MB_ICONERROR);
			return 1;
		}

		ShowWindow(hwnd, nShowCmd);
		UpdateWindow(hwnd);

		// 消息循环
		MSG msg;
		while (GetMessage(&msg, NULL, 0, 0))
		{
			TranslateMessage(&msg);
			DispatchMessage(&msg);
		}
		return (int)msg.wParam;
	}



void HandleSizeChange(HWND hWnd,UINT state, int width, int height)
{
	const wchar_t* stateStr = L"Unknown";
	switch (state)
	{
	case SIZE_MAXIMIZED:stateStr = L"MAXIMIZED"; break;
	case SIZE_MINIMIZED:stateStr = L"MINIMIZED"; break;
	case SIZE_RESTORED:stateStr = L"RESTORED"; break;

	}

	wchar_t buf[256];
	swprintf_s(buf, 256, L"State: %ls, Size: %dx%d\n", stateStr, width, height);
	OutputDebugStringW(buf);

	if (state != SIZE_MINIMIZED)
	{
		InvalidateRect(hWnd, NULL, TRUE);
	}

}
```
