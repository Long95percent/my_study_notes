实验3：
- 获取窗口句柄自动获取鼠标位置进行点击
![[Pasted image 20250916110939.png]]
```
#include<windows.h>
#include<iostream>

int main()
{
//不显示黑窗口
	HWND console = GetConsoleWindow();
	ShowWindow(console, SW_HIDE);

	Sleep(3000);


	HWND hwnd = FindWindow(NULL, L"mouse click");
	if (!hwnd) {
		MessageBox(NULL, L"未找到窗口，请先运行target_window.exe", L"错误", MB_ICONERROR);
		return -1;
	}

	HWND hButton = FindWindowEx(hwnd, NULL, L"Button", L"click me");
	if (!hButton) {
		MessageBox(NULL, L"未找到按钮", L"错误", MB_ICONERROR);
		return -1;
	}

	RECT rect;
	GetWindowRect(hButton, &rect);
	int centerX = (rect.left + rect.right) / 2;
	int centerY = (rect.top + rect.bottom) / 2;

	SetCursorPos(centerX, centerY);
	Sleep(200);

	mouse_event(MOUSEEVENTF_LEFTDOWN, 0, 0, 0, 0);
	Sleep(50);
	mouse_event(MOUSEEVENTF_LEFTUP, 0, 0, 0, 0);

	std::cout << "鼠标已点击按钮！" << std::endl;
	return 0;

}
```

实验4：自动点击记事本编辑区
![[Pasted image 20250916201923.png]]
- 注意WIN11已经不叫Edit了，直接用Edit找不到
- 使用以下代码查找窗口子控件的实际类名
```
BOOL CALLBACK EnumChildProc(HWND hwnd, LPARAM lParam)
{
    wchar_t className[256];
    GetClassName(hwnd, className, 256);
    wchar_t title[256];
    GetWindowText(hwnd, title, 256);

    std::wcout << L"Child HWND=" << hwnd 
               << L" Class=" << className 
               << L" Title=" << title << std::endl;

    return TRUE; // 继续枚举
}

int main()
{
    HWND hwndNotepad = FindWindow(L"Notepad", NULL);
    if (!hwndNotepad) {
        std::cout << "Cannot find Notepad window." << std::endl;
        return 1;
    }

    EnumChildWindows(hwndNotepad, EnumChildProc, 0);
    return 0;
}
```

实验5：模拟窗口键盘输入
根据文件里的原代码，无法找到“Edit”
查看notepad.exe的位置，发现是`C:\Program Files\WindowsApps\...` ，这是新版 AppX/UWP 版，不会有 Edit 控件。
![[Pasted image 20250919163709.png]]
由于版本问题，没有Notepad软件有win32 Edit控件，所以使用其它方法
尝试过经典版 Notepad 并用 `FindWindow` + `EnumChildWindows` 仍然找不到 `Edit`
![[Pasted image 20250919170353.png]]
```
#include <Windows.h>
#include <iostream>

int main()
{
    // 找 Notepad 主窗口（类名 Notepad）
    HWND hwnd = FindWindow(L"Notepad", NULL);
    if (!hwnd) {
        std::cout << "Not find Notepad!\n";
        return 0;
    }

    // 将 Notepad 置为前台
    DWORD tid = GetWindowThreadProcessId(hwnd, NULL);
    AttachThreadInput(GetCurrentThreadId(), tid, TRUE);
    SetForegroundWindow(hwnd);
    AttachThreadInput(GetCurrentThreadId(), tid, FALSE);
    Sleep(50); // 等待前台

    // 模拟按键 'A'
    INPUT ip[2] = {};
    ip[0].type = INPUT_KEYBOARD;
    ip[0].ki.wVk = 0x41; // 'A'
    ip[1] = ip[0];
    ip[1].ki.dwFlags = KEYEVENTF_KEYUP;

    SendInput(2, ip, sizeof(INPUT));

    std::cout << "Sent 'A' successfully.\n";
    return 0;
}

```

实验6：同时模拟键盘和鼠标输入
```
#include <Windows.h>
#include <iostream>
#include <string>

void SendText(const std::string& text)
{
    for (char ch : text) {
        SHORT vk = VkKeyScanA(ch);
        if (vk == -1) continue;

        INPUT inputs[2] = {};
        // 按下
        inputs[0].type = INPUT_KEYBOARD;
        inputs[0].ki.wVk = LOBYTE(vk);
        // 释放
        inputs[1].type = INPUT_KEYBOARD;
        inputs[1].ki.wVk = LOBYTE(vk);
        inputs[1].ki.dwFlags = KEYEVENTF_KEYUP;

        if (HIBYTE(vk) & 1)
        {
            INPUT shiftDown = { INPUT_KEYBOARD };
            shiftDown.ki.wVk = VK_SHIFT;

            INPUT shiftUp = { INPUT_KEYBOARD };   // ✅ 新增定义
            shiftUp.ki.wVk = VK_SHIFT;            // ✅ 指定虚拟键码
            shiftUp.ki.dwFlags = KEYEVENTF_KEYUP; // ✅ 设置成抬起

            INPUT all[4] = { shiftDown, inputs[0], inputs[1], shiftUp };
            SendInput(4, all, sizeof(INPUT));
        }
        else {
            SendInput(2, inputs, sizeof(INPUT));
        }
        Sleep(30);
    }
}

int main() {
    HWND hwnd = FindWindow(NULL, L"无标题 - 记事本");
    if (!hwnd)
    {
        std::cout << "未找到记事本" << std::endl;
        return 1;
    }

    SetForegroundWindow(hwnd);
    Sleep(300);

    SendText("Hello,world!");
    return 0;
}
```
